[docs]    def to_db(self):
        """This function takes the file path as parameter and pass it to
        EXCEL DB class as argument and call the class method excel_to_db
        Take the values given from class and print into screen
        Return a message if the operation has been successfully
        """
        file = filedialog.askopenfilename()
        if file:
            op = messagebox.askyesno(message='Subir excel directo?')
            if op:
                status = ExcelDB(file)
                self.DataGrid.insert(tk.END, status.excel_to_db())
                tk.messagebox.showinfo(message="Wait for 5 secs")
            else:
                messagebox.showinfo(message='No se subio ningun archivo')



[docs]    def view_data(self):
        """This function is responsible for load the values stored in a cvs file
            Takes the file.name as string and return a list with each rows proceeded
        """
        with open('solares1.csv', 'rt') as file:
            reader = csv.reader(file, delimiter=',', quotechar='|')
            for row in reader:
                self.DataGrid.insert(tk.END, row[1:])



[docs]    def query_criteria(self):
        """This function takes the values from the combobox and a entry field
        and pass its to SimpleQuery as argument
        The values will be processed as table.field name and value selected/gaven to query
        """
        self.DataGrid.delete(0, tk.END)
        # Column select
        q = self.Cbx.get()
        # Value to find
        dt = self.Entry.get()
        query_data = SimplyQuery(q, dt)
        self.DataGrid.insert(tk.END, query_data.make_query())
