import tkinter as tk
from tkinter import filedialog
from tkinter import ttk
from SolarApp.DataBases.db_conns import ImageConn
from SolarApp.img_to_file import ImageInput



[docs]class ImageLoader(tk.Tk):
	""" This class is responsible for the IMAGE INPUT FORM
		Return a form object (widgets)
	"""

	def __init__(self):
		tk.Tk.__init__(self)
		frame = tk.Frame(self, bg="yellow", width=380, height=420)
		file = filedialog.askopenfilename()
		val = ImageInput(file).image_proc()
		self.l1 = tk.Label(frame, text='Date', bg='orange')
		self.TypeVal = ttk.Combobox(frame, state="readonly")
		self.l2 = tk.Label(frame, text="Save in column", bg='orange')
		self.ChooseActions = ttk.Combobox(frame, state="readonly")
		""" The val variable could be set from img_to_file class"""
		self.TypeVal["values"] = val
		""" - - - Col names could be set here - - -"""
		self.ChooseActions["values"] = ["solar", "apellido", "nombre", "dni", "direccion", "telefono", "funcion"]
		self.button1 = tk.Button(frame, text='Apply', command=self.save_data)
		self.Datagrid = tk.Listbox(frame, bg="black", width=10, height=2, foreground="yellow")
		self.Datagrid.config(font=("Consolas", 12, "bold"))

		# pack Add-Ons
		self.l1.pack(side='top', fill='both')
		self.TypeVal.pack(side="top", fill="both")
		self.l2.pack(fill='both')
		self.ChooseActions.pack(side="top", fill="both")
		self.button1.pack()
		self.Datagrid.pack(fill="both")
		frame.pack(fill="both")


[docs]	def save_data(self):
		"""This function takes the value from TypeVal and Choose Actions and return the provides arguments
			for the ImageConn Class also calls to the procceded method of the class before mentioned
		"""
		t = self.TypeVal.get()
		c = self.ChooseActions.get()
		msg = ImageConn(c, t).procceded_to_db()
		self.Datagrid.insert(tk.END, msg)
