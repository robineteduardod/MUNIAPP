        """This function takes the f parameter from constructor

        file = self.f
        conn = sqlite3.connect(db)
        try:
            cur = conn.cursor()
            df = pd.read_excel(file, skiprows=0)
            df2 = df.to_csv('solares1.csv', index=True)
            time.sleep(5)
            with open('solares1.csv', 'rt') as file:
                reader = csv.DictReader(file, dialect='excel')
                for row in reader:
                    cur.execute('INSERT INTO personal (solar, apellido) Values(?,?);',
                                (row['solar'], row['datos']))
            conn.commit()
        except sqlite3.IntegrityError as e:
            conn.rollback()
        finally:
            conn.close()

        notify = 'All row inserted'
        return notify
