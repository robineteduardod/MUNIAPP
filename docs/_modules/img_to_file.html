[docs]	def image_proc(self):
		"""This function is responsible for proceed the ImageInput parameter
			Takes the image object provides by class to create a image string object
			Processed it and return its values into a compressed list
		"""
		image = self.img
		data_to_list = []
		try:
			# 	Need pytesseract or other Optical Chart Recognition
			img = pytesseract.image_to_string(image, lang='spa', nice=1)  # Data to format
			data_from_img = img.lstrip(" ").split(maxsplit=len(img))
			data_to_list = [x for x in data_from_img[15:250] if len(x) > 5 and x is not None]
			data_to_list.insert(0, 'datos:')
		except pytesseract.TesseractError as t:
			if t:
				tk.messagebox.showerror(message='Please load an image')
				quit()

		return data_to_list
