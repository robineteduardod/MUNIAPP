[docs]class InputManual(tk.Tk):
    """
    This class is responsible for the GUI form.Takes a tkinter objects and return a based- widget interface
    Also provides some functionality to proced the values stored in widgets fields
    """
    def __init__(self):
        """This the InputManual class constructor
            Takes a tkinter object and return a configured widget set up for receive the data input from user
        """
        tk.Tk.__init__(self)

        input_frame = tk.Frame(self, bg="red", width=380, height=420)  # main input_frame
        self.label1 = tk.Label(input_frame, text='Information Type')
        self.TypeInfo = ttk.Combobox(input_frame, state="readonly")
        #  Column values could be settings from here
        self.TypeInfo["values"] = ["solar", "apellido", 'codigo', "nombre", "dni", "direccion", "telefono", "funcion"]
        self.label2 = tk.Label(input_frame, text='Please write your info data')
        self.entry1 = tk.Entry(input_frame)
        self.label1.pack()
        self.TypeInfo.pack()
        self.label2.pack()
        self.entry1.pack()

        input_frame.pack(fill="both")

        input_buttons_bar = tk.Frame(self)  # A button bar
        self.button1 = tk.Button(input_buttons_bar, text='Add a record', command= self.manual_query)
        self.button1.pack()
        input_buttons_bar.pack(fill="both")
        pane = tk.Frame(self)
        self.Info = tk.Label(pane, text='Remember choose the right column for the right data'+'\n')
        self.Info.pack()
        pane.pack()


[docs]    def manual_query(self):
        """ This function takes the values stored in TypeInfo, entry variables provides by ttk combobox
            and tk.EntryField.
            return the provides arguments for build the ByHandsInput objectd and process it calling to
            the handswritter_input
        """
        col = self.TypeInfo.get()
        data = self.entry1.get()
        if ByHandsInput(col, data).handswritter_input() == 1:
            tk.messagebox.showerror(message='CHOOSE A COLUMN')
        elif data == '':
            tk.messagebox.showerror(message='A VALUE MUST BE INSERTED')
